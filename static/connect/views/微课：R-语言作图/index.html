# ggplot2 -----------------------------------------------------------------

gdp <- readRDS('https://pzhao.org/data/gdp.RDS')

p1 <- ggplot(gdp, aes(x = country, y = gdp, fill = gdp)) + 
  geom_col()

p1

p1 + scale_y_log10()

p1 + scale_y_log10() + coord_polar()

p1 + scale_y_log10() + coord_polar() +
  scale_fill_gradientn(colors=c("steelblue", "lightgreen", "orange", "red","darkred", "brown")) +
  geom_col(aes(y=I(10^4)), width=1, fill='white', alpha = .2) +
  geom_col(aes(y=I(10^3)), width=1, fill='white', alpha = .2) +
  geom_col(aes(y=I(10^2)), width=1, fill='white', alpha = .2) +
  geom_text(aes(label=country, y = gdp + 100000), size = 6) +
  annotate("text", x = 21, y = c(100, 1000, 10000, 100000), color = "white",
           label=c("百亿", "千亿", "万亿", "十万亿")) +
  theme_void() +
  theme(legend.position="none")

# basic R -----------------------------------------------------------------

barplot(gdp$gdp)

barplot(gdp$gdp, 
        names.arg = gdp$country, 
        las = 2, 
        col = rainbow(nrow(gdp)), 
        log = 'y')
box()

# plotly ------------------------------------------------------------------

library(plotly)
plot_ly(x = gdp$country, y = gdp$gdp)
ggplotly(p1)

# rgl ---------------------------------------------------------------------

heart <- readRDS('https://pzhao.org/data/heart.RDS')
library(rgl)
persp3d(heart$z, heart$x, heart$y, col = "red")
demo("ChinaHeart3D", package = "fun")
